# react-slide-tabs

```js
NONE
```

## usage:
```jsx
// import ReactSlideTabs from 'react-slide-tabs'

class App extends React.Component{
  render(){
    return (
      <div className="hello-react-slide-tabs">
        <ReactSlideTabs>
          <div className="bd" menu="red">
            Read tab contet...
          </div>
          <div className="bd" menu="blue">
            Blue tab contet...
          </div>
          <div className="bd" menu="yellow">
            Yellow tab contet...
          </div>
        </ReactSlideTabs>
    </div>
    );
  }
}
```
